    <script>
        function showLoading() {
            document.getElementById('loading').style.display = 'block';
            document.getElementById('results').style.opacity = '0.5';
        }

        function hideLoading() {
            document.getElementById('loading').style.display = 'none';
            document.getElementById('results').style.opacity = '1';
        }

        function parseCardInput(input) {
            return input.split(',').map(card => card.trim().toUpperCase()).filter(card => card);
        }

        function getSuitClass(suit) {
            return suit-${suit};
        }

        function formatCardDisplay(card) {
            if (!card) return '-';
            const suit = card.slice(-1);
            const rank = card.slice(0, -1);
            return ${rank}<span class="card-suit ${getSuitClass(suit)}">${suit}</span>;
        }

        async function analyzeGame() {
            showLoading();
            
            const hand = parseCardInput(document.getElementById('hand').value);
            const playedCards = parseCardInput(document.getElementById('playedCards').value);
            const currentRound = parseInt(document.getElementById('currentRound').value);
            const currentLeader = document.getElementById('currentLeader').value;

            try {
                const response = await fetch('/analyze', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        hand: hand,
                        played_cards: playedCards,
                        round: currentRound,
                        leader: currentLeader,
                        tricks_won: {P1: 0, P2: 0, P3: 0, P4: 0},
                        void_info: {P1: [], P2: [], P3: [], P4: []}
                    })
                });

                const data = await response.json();

                if (data.error) {
                    throw new Error(data.error);
                }

                // Display best card
                document.getElementById('bestCardDisplay').innerHTML = formatCardDisplay(data.best_card);
                document.getElementById('bestCardReason').textContent = data.best_reason;

                // Display hand analysis
                document.getElementById('handAnalysisContent').innerHTML = `
                    Total cards: ${data.hand_analysis.total_cards}<br>
                    Spades: ${data.hand_analysis.spades_count}<br>
                    High cards (J+): ${data.hand_analysis.high_cards_count}
                `;

                // Display opponent predictions
                document.getElementById('opponentPredictionsContent').innerHTML = 
                    data.opponent_predictions.replace(/\n/g, '<br>');

                // Display survival analysis
                document.getElementById('survivalAnalysisContent').innerHTML = 
                    data.survival_analysis.replace(/\n/g, '<br>');

            } catch (error) {
                alert('Error: ' + error.message);
            } finally {
                hideLoading();
            }
        }

        async function analyzeSpecificCard() {
            const specificCard = document.getElementById('specificCard').value.trim().toUpperCase();
            const hand = parseCardInput(document.getElementById('hand').value);
            const playedCards = parseCardInput(document.getElementById('playedCards').value);
            const currentRound = parseInt(document.getElementById('currentRound').value);

            if (!specificCard) {
                alert('Please enter a card to analyze');
                return;
            }

            if (!hand.includes(specificCard)) {
                alert('This card is not in your hand!');
                return;
            }

            showLoading();

            try {
                const response = await fetch('/analyze_card', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        card: specificCard,
                        hand: hand,
                        played_cards: playedCards,
                        round: currentRound,
                        void_info: {P1: [], P2: [], P3: [], P4: []}
                    })
                });

                const data = await response.json();

                if (data.error) {
                    throw new Error(data.error);
                }

                document.getElementById('specificCardAnalysisContent').innerHTML = `
                    <div class="card-display">${formatCardDisplay(data.card)}</div>
                    <p><strong>Value:</strong> ${data.value}</p>
                    <p><strong>Suit:</strong> <span class="${getSuitClass(data.suit)}">${data.suit}</span></p>
                    <p><strong>Confidence:</strong> ${data.confidence}%</p>
                    <div class="confidence-bar">
                        <div class="confidence-fill" style="width: ${data.confidence}%"></div>
                    </div>
                    <div class="predictions">
                        ${data.predictions.map(pred => <div class="prediction-item">${pred}</div>).join('')}
                    </div>
                `;

            } catch (error) {
                alert('Error: ' + error.message);
            } finally {
                hideLoading();
            }
        }
